decision_planet_perma_food_boost = {
	owned_planets_only = yes
	sound = event_administrative_work
	icon = decision_resources

	# resources = {
	# 	category = decisions
	# 	cost = {
	# 	}
	# }

	potential = {
		owner = {
			is_regular_empire = yes
			is_lithoid_empire = no
			is_ai = no
		}
		NOR = {
			has_modifier = planet_growth_discouraged
			has_modifier = planet_population_control
			has_modifier = planet_food_boost
			has_planet_flag = has_perma_food_boost
		}
	}

	effect = {
		add_modifier = {
			modifier = "planet_food_boost"
		}
		hidden_effect = {
			add_deposit = perma_food_boost_deposit
			set_planet_flag = has_perma_food_boost
		}
	}
}

decision_planet_remove_perma_food_boost = {
	owned_planets_only = yes
	sound = event_administrative_work
	icon = decision_resources

	potential = {
		owner = {
			is_regular_empire = yes
			is_lithoid_empire = no
			is_ai = no
		}
		AND = {
			has_planet_flag = has_perma_food_boost
			has_deposit = perma_food_boost_deposit
		}
	}

	effect = {
		remove_modifier = planet_food_boost
		hidden_effect = {
			remove_deposit = perma_food_boost_deposit
			remove_planet_flag = has_perma_food_boost
		}
	}
}
